# Lancement bdd
	docker container run -d --name mongodb mongo

# Création Dockerfile "seed_mongo"
	FROM python:3.10-alpine

	RUN pip install pymongo

	COPY seeder.py /

	CMD ["python3", "seeder.py"]

# Lancement seeder
	docker container run --name seed_mongo_c --link mongodb:mongodb seed_mongo

# Création Dockerfile "app"
	FROM python:3.10-alpine

	RUN pip install uvicorn pymongo fastapi

	COPY model_egg.py /

	COPY immat_parser.py /

	COPY app.py /

	EXPOSE 3000

	CMD ["uvicorn", "app:app", "--host", "0.0.0.0", "--port", "3000"]

# Lancement app
	docker container run --name app_c -p 3000:3000 --link mongodb:mongodb app

	# On peut aller sur localhost:3000 et voir l'api !

### Dans l'ordre (sans network) :
	docker container run -d --name mongodb mongo
	docker container run --name seed_mongo_c --link mongodb:mongodb seed_mongo
	docker container run --name app_c -p 3000:3000 --link mongodb:mongodb app
